//Pedaço do código JavaScript que inicializa uma Promise:
  
  let minhaPromessa = new Promise((resolve, reject) => {

    // aqui executamos algo demorado

    if (/* tudo deu certo */) {
        resolve("Funcionou!");
    }
    else {
        reject("Deu erro...");
    }
});


//Verificar o resultado da execução

  minhaPromessa
      .then(mensagem => console.log(mensagem))
      .catch(erro => console.log(erro));

//Usa-se os métodos then e catch para capturar o resolve e o reject oriundo da promisse e dentro deles é necessário passar
  uma function ou Arrow function com o comando desejado.
